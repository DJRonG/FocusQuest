// Keep track of activities and score
let score = 0;
const scoreDisplay = document.getElementById("scoreValue");
const activityList = document.getElementById("activityList");
const newActivityInput = document.getElementById("newActivity");

// Add new activity + update score
function addActivity() {
  const activityText = newActivityInput.value.trim();

  if (activityText !== "") {
    const listItem = document.createElement("li");
    listItem.textContent = activityText;
    activityList.appendChild(listItem);

    // Basic scoring logic
    // Each activity adds +10 points
    score += 10;
    scoreDisplay.textContent = score;

    // Optional: encourage message
    if (score % 50 === 0) {
      alert("ðŸ”¥ Great job! Youâ€™ve reached a new focus level!");
    }

    newActivityInput.value = "";
  }
}

// Optional: Keyboard shortcut (Enter key)
newActivityInput.addEventListener("keypress", function (e) {
  if (e.key === "Enter") {
    addActivity();
  }
});
